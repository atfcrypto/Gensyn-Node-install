# Remove old Docker installations

# Install dependencies

# Add Docker’s official GPG key
curl -fsSL https://download.docker.com/linux/ubuntu/gpg | sudo gpg --dearmor -o /usr/share/keyrings/docker-archive-keyring.gpg

# Set up the stable Docker repository
echo \
  "deb [arch="$(dpkg --print-architecture)" signed-by=/etc/apt/keyrings/docker.gpg] https://download.docker.com/linux/ubuntu \
  "$(. /etc/os-release && echo "$VERSION_CODENAME")" stable" | \
  sudo tee /etc/apt/sources.list.d/docker.list > /dev/null


# Install Docker
sudo apt install docker-ce docker-ce-cli containerd.io -y

# Verify Docker installation
sudo docker --version

# Start and enable Docker
sudo systemctl enable docker
sudo systemctl start docker

# Run Docker without sudo (optional)
sudo usermod -aG docker $USER

# Test Docker installation

5) Login
Wait for Waiting for userData.json to be created... in logs



Open the login page in your browser:

Local PC: http://localhost:3000/

VPS users: If you don't receive the OTP email when logging in on port 3000, you must forward the port. (See Step 3 below)

⚠️ If you can't login or don't receive the email code, Forward the port (ONLY FOR VPS USERS):

On your local PC, open PowerShell

Enter the command below (replace Server_IP with your VPS IP and SSH_PORT with your VPS port, e.g., 22):

bash
Copy
Edit
ssh -L 3000:localhost:3000 root@Server_IP -p SSH_PORT
Example
bash
Copy
Edit
ssh -L 3000:localhost:3000 root@137.12.0.0 -p 22
After entering this command, type YES and provide your VPS password.

Important:

Run the above command in your local Windows PowerShell cmd and not on your VPS terminal.

This command tunnels the port, allowing you to open http://localhost:3000/ in your browser and log in.
